# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def longestConsecutive(self, root: Optional[TreeNode]) -> int:
        self.path = []
        self.result = 0
      
        self.traverse(root, None, 0)
      
        return self.result
  
    def traverse(self, root: Optional[TreeNode], parent: Optional[TreeNode], curCount: int):
        if not root:
            return
        if not parent:
            curCount = 1
        else:
            if parent.val + 1 == root.val:
                curCount += 1
            else:
                curCount = 1
        self.result = max(self.result, curCount)
      
        self.traverse(root.left, root, curCount)
        self.traverse(root.right,root, curCount)
