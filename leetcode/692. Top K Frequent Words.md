```python
class Solution:
    def topKFrequent(self, words: List[str], k: int) -> List[str]:
        result = []
        countMap = defaultdict(int)
        for word in words:
            countMap[word] += 1
      
        heap = []
        for word, v in countMap.items():
            heapq.heappush(heap, [-v, word])
          
        for i in (i for i in range(k) if heap):
            _, word = heapq.heappop(heap)
            result.append(word)
          
        return result
```
